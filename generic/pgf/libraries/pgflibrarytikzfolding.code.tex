\newdimen\pentagonlinelength
\pentagonlinelength=2cm

\def\pentagon#1#2#3#4#5#6{%
  \begin{scope}[xshift=.5\pentagonlinelength,yshift=0.6881909602355868\pentagonlinelength]
    #1
  \end{scope}
  \begin{scope}[shift={(108:\pentagonlinelength)},shift={(36:\pentagonlinelength)},rotate=-36]
    #2
  \end{scope}
  \begin{scope}[xshift=\pentagonlinelength,shift={(72:\pentagonlinelength)},rotate=-108]
    #3
  \end{scope}
  \begin{scope}[xshift=\pentagonlinelength,rotate=180]
    #4
  \end{scope}
  \begin{scope}[rotate=108]
    #5
  \end{scope}
  \begin{scope}[shift={(108:\pentagonlinelength)},rotate=36]
    #6
  \end{scope}
}

\def\foldline{\draw[help lines](0,0) -- (\pentagonlinelength,0pt);}
\def\cutline{\draw(0,0) -- (\pentagonlinelength,0pt);}
\def\ear{\draw[help lines](0,0) -- (\pentagonlinelength,0pt);\draw(0,0) --
(.5\pentagonlinelength,.2\pentagonlinelength) --(\pentagonlinelength,0pt);}
