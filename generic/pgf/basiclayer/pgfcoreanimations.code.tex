% Begin and end an animation scope
%
% #1 = a scope name
%
% Description:
%
% The graphic objects created inside an animation scope can be the
% target of an animation command (see the "whom" key). The scope name
% can be reused and is set globally.

\def\pgfanimationscope#1{%
  \pgfsys@begin@animation@scope%
    \expandafter\xdef\csname pgf@anim@scope@name@#1\endcsname{\pgfsys@animationscope@id}}
\def\endpgfanimationscope{%
  \pgfsys@end@animation@scope%
}

\let\startpgfanimationscope\pgfanimationscope
\let\stoppgfanimationscope\endpgfanimationscope

\def\pgfanimateattribute#1#2{\pgferror{You need to say \string\usepgfmodule{animations} to use animations}}


\endinput
